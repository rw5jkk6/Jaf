from flask import Flask, render_template, request
import math
from scipy.stats import norm

app = Flask(__name__)

def calculate_greeks(S, K, T_days, r, sigma, option_type='call'):
    """
    ブラック・ショールズ・モデルによるオプション価格とギリシャ指標の計算
    """
    T = T_days / 250
    # ゼロ除算やマイナス期間の防止
    if T <= 0: T = 0.00001
    if sigma <= 0: sigma = 0.0001

    try:
        d1 = (math.log(S / K) + (r + 0.5 * sigma**2) * T) / (sigma * math.sqrt(T))
        d2 = d1 - sigma * math.sqrt(T)
        
        N_d1 = norm.cdf(d1)
        N_d2 = norm.cdf(d2)
        n_d1 = norm.pdf(d1)
        
        if option_type == 'call':
            price = S * N_d1 - K * math.exp(-r * T) * N_d2
            delta = N_d1
            theta = (- (S * n_d1 * sigma) / (2 * math.sqrt(T)) - r * K * math.exp(-r * T) * N_d2) / 365
        else:
            price = K * math.exp(-r * T) * norm.cdf(-d2) - S * norm.cdf(-d1)
            delta = N_d1 - 1
            theta = (- (S * n_d1 * sigma) / (2 * math.sqrt(T)) + r * K * math.exp(-r * T) * norm.cdf(-d2)) / 365

        gamma = n_d1 / (S * sigma * math.sqrt(T))
        vega = (S * math.sqrt(T) * n_d1) / 100

        return {
            "price": price, 
            "delta": delta, 
            "gamma": gamma, 
            "theta": theta, 
            "vega": vega
        }
    except Exception as e:
        print(f"Calculation Error: {e}")
        return None

@app.route('/', methods=['GET', 'POST'])
def index():
    # デフォルトの入力値
    default_inputs = [
        {'S': 10000, 'K': 10500, 'T': 30, 'r': 0.01, 'sigma': 0.3},
        {'S': 10000, 'K': 10500, 'T': 30, 'r': 0.01, 'sigma': 0.4}
    ]
    
    data = []
    
    for i in range(2):
        if request.method == 'POST':
            # フォームから値を取得。取得できない場合はデフォルト値を使用
            try:
                s_val = float(request.form.get(f'S_{i}') or default_inputs[i]['S'])
                k_val = float(request.form.get(f'K_{i}') or default_inputs[i]['K'])
                t_val = float(request.form.get(f'T_{i}') or default_inputs[i]['T'])
                r_val = float(request.form.get(f'r_{i}') or default_inputs[i]['r'])
                sigma_val = float(request.form.get(f'sigma_{i}') or default_inputs[i]['sigma'])
            except ValueError:
                # 数値以外が入力された場合のフォールバック
                s_val, k_val, t_val, r_val, sigma_val = default_inputs[i].values()

            current_inputs = {'S': s_val, 'K': k_val, 'T': t_val, 'r': r_val, 'sigma': sigma_val}
            results = {
                'call': calculate_greeks(s_val, k_val, t_val, r_val, sigma_val, 'call'),
                'put': calculate_greeks(s_val, k_val, t_val, r_val, sigma_val, 'put')
            }
        else:
            # 初期表示(GET)時
            current_inputs = default_inputs[i]
            results = None
            
        data.append({'inputs': current_inputs, 'results': results})
            
    return render_template('index.html', data=data)

if __name__ == '__main__':
    # debug=True にするとエラー時にブラウザ上で詳細が確認できます
    app.run(debug=True, port=5000)
